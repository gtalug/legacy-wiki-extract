function(doc) {
	if (doc.fn) {
		var name = (doc.fn['given-name'] + " " + doc.fn['family-name'])
		emit(name, doc);
	}
	else if (doc.org) {
		emit(doc.org['organization-name'], doc)
	}
};

map.js 
function(doc) {
	if (doc.fn && doc.email) {
		for (var idx in doc.email) {
			emit(doc.email[idx]['value'], doc)
		}
	}
